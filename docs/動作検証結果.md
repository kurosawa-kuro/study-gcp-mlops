```bash
MEILI_URL=$(gcloud run services describe meili-search --project=mlops-dev-a --region=asia-northeast1 --format='value(status.url)')

  -d '{"filterableAttributes":["rent","walk_min","age_years","layout","pet_ok"],"searchableAttributes":["title","city","ward","layout"]}'

# 2) documents PUT — **rent / walk_min / age_years / pet_ok は numeric / bool 必須**
#    bq query --format=json は int64 を string に変換する仕様で、Meili filter `rent <= 150000` が壊れる。

curl -sS -X PUT "$MEILI_URL/indexes/properties/documents" \
  -H "Authorization: Bearer $TOKEN" -H "Content-Type: application/json" \
  --data-binary @docs_numeric.json  # rent:120000 (int), pet_ok:true (bool) 等
# Expect: 202 → /tasks/N で status=succeeded を確認
```
